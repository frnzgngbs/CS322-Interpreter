BEGIN CODE
    INT rows = 5

    INT currentRow = 1
    INT spaces = rows - 1
    INT hashes = 1

    WHILE (currentRow <= rows)
    BEGIN WHILE
         INT spaceCount = 0
         WHILE (spaceCount < spaces)
         BEGIN WHILE
            DISPLAY: " "
            spaceCount = spaceCount + 1
         END WHILE

         INT hashCount = 0
         WHILE (hashCount < hashes)
         BEGIN WHILE
             DISPLAY: "#"
             hashCount = hashCount + 1
         END WHILE

         DISPLAY: $
         spaces = spaces - 1
         hashes = hashes + 2
         currentRow = currentRow + 1
    END WHILE

END CODE
